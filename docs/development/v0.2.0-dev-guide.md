```typescript
// src/components/ProjectHome/StatsRow.tsx
import { useEffect, useState } from 'react'
import { message } from 'antd'

interface StatsRowProps {
  projectId: number
  milestones: Milestone[] | null
  onViewSummary: () => void
  onGenerateSummary: () => void
}

export default function StatsRow({ projectId, milestones, onViewSummary, onGenerateSummary }: StatsRowProps) {
  const [fileCount, setFileCount] = useState(0)
  const [pendingCount, setPendingCount] = useState(0)
  const [hasSummary, setHasSummary] = useState(false)

  useEffect(() => {
    loadStats()
  }, [projectId])

  const loadStats = async () => {
    const filesResult = await fileService.list(projectId)
    if (filesResult.success && filesResult.data) {
      setFileCount(filesResult.data.length)
      setPendingCount(filesResult.data.filter(f => !f.is_analyzed).length)
    }

    const summaryResult = await fileService.getSummary(projectId)
    setHasSummary(summaryResult.success)
  }

  const nextMilestone = getNextMilestone(milestones)

  return (
    <div className="summary-row">
      {/* 文件数量 */}
      <div className="summary-card">
        <div className="summary-card-header">
          <div className="summary-card-icon sc-icon-blue">📁</div>
          <span className="summary-card-label">文件数量</span>
        </div>
        <span className="summary-card-value">{fileCount}</span>
        <span className="summary-card-footer">个项目文件</span>
      </div>

      {/* 里程碑 */}
      <div className="summary-card" style={{ cursor: 'pointer' }} onClick={() => openMilestoneModal()}>
        <div className="summary-card-header">
          <div className="summary-card-icon" style={{ background: 'var(--color-accent-light)', color: 'var(--color-accent)' }}>⏱</div>
          <span className="summary-card-label">里程碑</span>
        </div>
        <span className="summary-card-value" style={{ fontSize: '16px', lineHeight: 1.3 }}>
          {nextMilestone?.title || '—'}
        </span>
        <span className="summary-card-footer">
          {nextMilestone ? `下一步 · ${formatDate(nextMilestone.date)}` : '暂无里程碑'}
        </span>
      </div>

      {/* 待处理 */}
      <div className="summary-card">
        <div className="summary-card-header">
          <div className="summary-card-icon sc-icon-red">⏳</div>
          <span className="summary-card-label">待处理</span>
        </div>
        <span className="summary-card-value">{pendingCount}</span>
        <span className="summary-card-footer">个文件待分析</span>
      </div>

      {/* AI摘要 */}
      <div className="summary-card">
        <div className="summary-card-header">
          <div className="summary-card-icon sc-icon-purple">🤖</div>
          <span className="summary-card-label">AI 摘要</span>
        </div>
        <div className="summary-card-action">
          <button className="btn btn-secondary btn-sm" onClick={onViewSummary}>查看摘要</button>
          <button className="btn btn-primary btn-sm" onClick={onGenerateSummary}>生成/更新</button>
        </div>
      </div>
    </div>
  )
}

function getNextMilestone(milestones: Milestone[] | null): Milestone | null {
  if (!milestones || milestones.length === 0) return null
  const now = new Date()
  return milestones.find(m => new Date(m.date) > now) || null
}
```

```typescript
import SummaryCard from './SummaryCard'

interface FeatureCardsProps {
  project: Project
  selectedCategory: string | null
  allFiles: FileRecord[]
}

type StageGroup = 'presale' | 'progress' | 'closed'

function getStageGroup(sidebarItem: string | null, projectStage: string): StageGroup {
  if (!sidebarItem || sidebarItem === '所有文件') {
    if (projectStage === '售前') return 'presale'
    if (projectStage === '关闭') return 'closed'
    return 'progress'
  }
  if (sidebarItem === '售前') return 'presale'
  if (sidebarItem === '关闭') return 'closed'
  return 'progress'
}

export default function FeatureCards({ project, selectedCategory, allFiles }: FeatureCardsProps) {
  const stageGroup = useMemo(
    () => getStageGroup(selectedCategory, project.current_stage),
    [selectedCategory, project.current_stage]
  )

  return (
    <div className="feature-row">
      {stageGroup === 'presale' && (
        <>
          <ProjectInfoCard project={project} />
          <ContractCard files={allFiles} />
          <EvaluationCard project={project} />
        </>
      )}
      {stageGroup === 'progress' && (
        <>
          <RequirementCard project={project} />
          <IssueCard project={project} />

        <>
          <OpportunityCard project={project} />

        </>
```

### 2.4 里程碑弹窗实现

```typescript
// src/components/ProjectHome/MilestoneModal.tsx
import { Modal, Timeline, Tag } from 'antd'
import type { Milestone } from '../../types'

interface MilestoneModalProps {
  open: boolean
  onClose: () => void
  milestones: Milestone[]
}

export default function MilestoneModal({ open, onClose, milestones }: MilestoneModalProps) {
  const now = new Date()

  const getMilestoneStatus = (milestone: Milestone, index: number) => {
    const date = new Date(milestone.date)
    if (date < now) return 'done'
    // 第一个未过期的是"下一步"
    const nextIndex = milestones.findIndex(m => new Date(m.date) > now)
    if (index === nextIndex) return 'next'
    return 'pending'
  }

  const isKeyMilestone = (title: string) => {
    return /合同|上线|验收/.test(title)
  }

  return (
    <Modal
      title="里程碑时间轴"
      open={open}
      onCancel={onClose}
      footer={null}
      width={640}
    >
      <Timeline
        items={milestones.map((m, index) => {
          const status = getMilestoneStatus(m, index)
          const isKey = isKeyMilestone(m.title)

          return {
            color: status === 'done' ? 'indigo' : status === 'next' ? 'amber' : 'gray',
            children: (
              <div style={{ display: 'flex', justifyContent: 'space-between' }}>
                <span style={{
                  textDecoration: status === 'done' ? 'line-through' : 'none',
                  color: status === 'pending' ? 'var(--text-placeholder)' : 'var(--text-primary)',
                  fontWeight: status === 'next' ? 600 : 400,
                }}>
                  {m.title}
                  {isKey && <span style={{ marginLeft: 4 }}>★</span>}
                  {status === 'next' && <Tag color="amber" style={{ marginLeft: 8 }}>下一步</Tag>}
                </span>
                <span style={{
                  color: status === 'next' ? 'var(--color-accent)' : 'var(--text-placeholder)',
                  fontWeight: status === 'next' ? 500 : 400,
                  fontFeatureSettings: '"tnum"',
                }}>
                  {formatDate(m.date)}
                </span>
              </div>
            )
          }
        })}
      />
    </Modal>
  )
}
```

### 4.2 HandoverService 核心实现

```typescript
// electron/services/handover-service.ts
import JSZip from 'jszip'
import fs from 'fs/promises'
import path from 'path'
import { getProject, listProjects } from '../database/projects'
import { listFiles } from '../database/files'
import { resolveProjectPathForProject } from '../utils/project-path'
import { getSetting } from '../database/settings'

export interface HandoverExportParams {
  projectId: number
  mode: 'full' | 'selective'
  selectedFiles?: string[]      // 仅selective模式
  handoverNote?: string
}

export class HandoverService {
  /**
   * 导出转交包
   */
  static async exportHandover(params: HandoverExportParams): Promise<Buffer> {
    const { projectId, mode, selectedFiles, handoverNote } = params

    // 1. 获取项目数据
    const project = getProject(projectId)
    if (!project) throw new Error('项目不存在')

    // 2. 获取项目文件夹路径
    const projectPath = await resolveProjectPathForProject(project)
    if (!projectPath) throw new Error('项目文件夹不存在')

    // 3. 获取文件列表
    const allFiles = listFiles(projectId)
    const filesToExport = mode === 'full'
      ? allFiles
      : allFiles.filter(f => selectedFiles?.includes(f.filename))

    // 4. 创建ZIP
    const zip = new JSZip()

    // 5. 添加handover.json
    const handoverJSON = {
      format_version: '1.0',
      type: mode,
      project: {
        name: project.name,
        category_type: project.category_type,
        current_stage: project.current_stage,
        created_at: project.created_at,
        milestones: project.milestones ? JSON.parse(project.milestones) : [],
        metadata: project.metadata ? JSON.parse(project.metadata) : {},
      },
      files: filesToExport.map(f => ({
        filename: f.filename,
        relative_path: `files/${f.category || '未分类'}/${f.filename}`,
        category: f.category,
        subcategory: f.subcategory,
        stage: f.stage,
        file_type: f.file_type,
        file_size: f.file_size,
        content_extracted: f.content_extracted,
        signature_status: f.signature_status,
        ai_summary: f.ai_summary,
        ai_key_info: f.ai_key_info ? JSON.parse(f.ai_key_info) : null,
      })),
      handover_note: handoverNote || '',
      exported_at: new Date().toISOString(),
      exported_by: `PMAer v${app.getVersion()}`,
    }
    zip.file('handover.json', JSON.stringify(handoverJSON, null, 2))

    // 6. 添加项目摘要
    try {
      const summaryPath = path.join(projectPath, '.ai', 'project-summary.md')
      const summary = await fs.readFile(summaryPath, 'utf-8')
      zip.file('project-summary.md', summary)
    } catch {
      // 摘要不存在，跳过
    }

    // 7. 添加关键信息
    try {
      const infoPath = path.join(projectPath, '.ai', 'project-info.md')
      const info = await fs.readFile(infoPath, 'utf-8')
      zip.file('key-info.md', info)
    } catch {
      // 信息不存在，跳过
    }

    // 8. 添加转交说明
    if (handoverNote) {
      zip.file('context.md', handoverNote)
    }

    // 9. 添加文件
    for (const file of filesToExport) {
      const filePath = path.join(projectPath, file.filename)
      try {
        const content = await fs.readFile(filePath)
        const category = file.category || '未分类'
        zip.file(`files/${category}/${file.filename}`, content)
      } catch {
        // 文件不存在，跳过
      }
    }

    // 10. 生成ZIP Buffer
    return zip.generateAsync({ type: 'nodebuffer', compression: 'DEFLATE' })
  }

  /**
   * 导入转交包
   */
  static async importHandover(zipPath: string, projectName?: string): Promise<number> {
    // 1. 读取ZIP
    const zipData = await fs.readFile(zipPath)
    const zip = await JSZip.loadAsync(zipData)

    // 2. 读取handover.json
    const handoverJSONFile = zip.file('handover.json')
    if (!handoverJSONFile) throw new Error('无效的转交包：缺少handover.json')

    const handoverJSON = JSON.parse(await handoverJSONFile.async('text'))

    // 3. 校验格式版本
    if (handoverJSON.format_version !== '1.0') {
      throw new Error(`不支持的转交包格式版本：${handoverJSON.format_version}`)
    }

    // 4. 创建项目
    const projectDb = require('../database/projects')
    const finalName = projectName || handoverJSON.project.name
    const projectId = projectDb.createProject(
      finalName,
      handoverJSON.project.category_type,
      undefined,
      require('uuid').v4()
    )

    // 5. 创建项目文件夹
    const { createProjectDirectory, createStageFolders } = require('../utils/project-path')
    const projectPath = await createProjectDirectory(projectId, finalName)

    // 6. 解压文件
    const filesFolder = zip.folder('files')
    if (filesFolder) {
      for (const [category, categoryFolder] of Object.entries(filesFolder)) {
        if (categoryFolder instanceof JSZip) {
          for (const [filename, file] of Object.entries(categoryFolder)) {
            if (file instanceof JSZip && !file.dir) {
              const content = await file.async('nodebuffer')
              await fs.writeFile(path.join(projectPath, filename), content)
            }
          }
        }
      }
    }

    // 7. 导入文件记录到数据库
    const fileDb = require('../database/files')
    for (const fileInfo of handoverJSON.files) {
      fileDb.createFile(projectId, {
        filename: fileInfo.filename,
        original_path: null,
        stored_path: path.join(projectPath, fileInfo.filename),
        category: fileInfo.category,
        subcategory: fileInfo.subcategory,
        stage: fileInfo.stage,
        file_type: fileInfo.file_type,
        file_size: fileInfo.file_size,
        content_extracted: fileInfo.content_extracted,
        is_analyzed: !!fileInfo.ai_summary,
        has_signature: fileInfo.signature_status === 'signed',
        signature_status: fileInfo.signature_status,
        ai_summary: fileInfo.ai_summary,
        ai_key_info: fileInfo.ai_key_info ? JSON.stringify(fileInfo.ai_key_info) : null,
      })
    }

    // 8. 导入里程碑
    if (handoverJSON.project.milestones?.length) {
      projectDb.updateProject(projectId, {
        milestones: JSON.stringify(handoverJSON.project.milestones),
      })
    }

    // 9. 导入元数据
    if (handoverJSON.project.metadata && Object.keys(handoverJSON.project.metadata).length) {
      projectDb.updateProject(projectId, {
        metadata: JSON.stringify(handoverJSON.project.metadata),
      })
    }

    return projectId
  }

  /**
   * 预览转交包
   */
  static async previewHandover(zipPath: string): Promise<any> {
    const zipData = await fs.readFile(zipPath)
    const zip = await JSZip.loadAsync(zipData)

    const handoverJSONFile = zip.file('handover.json')
    if (!handoverJSONFile) throw new Error('无效的转交包')

    return JSON.parse(await handoverJSONFile.async('text'))
  }

  /**
   * AI推荐文件
   */
  static async aiSelectFiles(projectId: number, description: string): Promise<{
    files: Array<{ filename: string; reason: string }>
    keyContext: string
  }> {
    const { getAIService } = require('../services/ai-service')
    const aiService = getAIService()

    if (!aiService.hasProviders()) {
      throw new Error('未配置AI服务')
    }

    // 获取项目文件列表
    const allFiles = listFiles(projectId)
    const fileList = allFiles.map(f =>
      `- ${f.filename} [${f.category || '未分类'}] ${f.ai_summary ? f.ai_summary.substring(0, 100) : ''}`
    ).join('\n')

    // 获取项目摘要
    const project = getProject(projectId)
    const projectSummary = project?.metadata
      ? JSON.parse(project.metadata).project_overview || ''
      : ''

    // 构建prompt
    const prompt = `你是PMAer项目转交助手。根据新同事的角色描述，从项目文件中选择最相关的文件。

## 新同事描述
${description}

## 项目文件列表（含分类信息）
${fileList}

## 项目摘要
${projectSummary}

请返回JSON格式：
{
  "selected_files": [
    {
      "filename": "文件名",
```

```typescript
// src/components/Handover/HandoverDialog.tsx
import { useState } from 'react'
import { Modal, Radio, Input, Button, Checkbox, message, Spin } from 'antd'
import { projectService } from '../../services/projectService'

interface HandoverDialogProps {
  open: boolean
  onClose: () => void
  projectId: number
  projectName: string
}

export default function HandoverDialog({ open, onClose, projectId, projectName }: HandoverDialogProps) {
  const [mode, setMode] = useState<'full' | 'selective'>('full')
  const [description, setDescription] = useState('')
  const [handoverNote, setHandoverNote] = useState('')
  const [loading, setLoading] = useState(false)
  const [aiFiles, setAiFiles] = useState<Array<{ filename: string; reason: string }>>([])
  const [selectedFiles, setSelectedFiles] = useState<string[]>([])

  const handleAiSelect = async () => {
    if (!description.trim()) {
      message.warning('请填写新同事描述')
      return
    }
    setLoading(true)
    try {
      const result = await projectService.aiSelectFiles(projectId, description)
      if (result.success && result.data) {
        setAiFiles(result.data.files)
        setSelectedFiles(result.data.files.map(f => f.filename))
      }
    } catch (err) {
      message.error('AI推荐失败，请手动选择文件')
    } finally {
      setLoading(false)
    }
  }

  const handleExport = async () => {
    setLoading(true)
    try {
      const result = await projectService.exportHandover({
        projectId,
        mode,
        selectedFiles: mode === 'selective' ? selectedFiles : undefined,
        handoverNote,
      })
      if (result.success && result.data) {
        message.success('转交包已生成')
        onClose()
      }
    } catch (err) {
      message.error('导出失败')
    } finally {
      setLoading(false)
    }
  }

  return (
    <Modal
      title="项目转交"
      open={open}
      onCancel={onClose}
      width={640}
      footer={[
        <Button key="cancel" onClick={onClose}>取消</Button>,
        <Button key="export" type="primary" loading={loading} onClick={handleExport}>
          导出 .pmaer.zip
        </Button>,
      ]}
    >
      <div style={{ marginBottom: 16 }}>
        <div style={{ marginBottom: 8, fontWeight: 500 }}>转交模式</div>
        <Radio.Group value={mode} onChange={e => setMode(e.target.value)}>
          <Radio value="full">整体转交 — 打包所有项目文件和信息</Radio>
          <br />
          <Radio value="selective">协作转交 — AI推荐相关文件，可手动调整</Radio>
        </Radio.Group>
      </div>

      {mode === 'selective' && (
        <>
          <div style={{ marginBottom: 16 }}>
            <div style={{ marginBottom: 8, fontWeight: 500 }}>新同事描述 *</div>
            <Input.TextArea
              rows={3}
              value={description}
              onChange={e => setDescription(e.target.value)}
              placeholder="例：张工负责前端开发，需要了解UI规范和接口文档"
            />
          </div>

          <Button onClick={handleAiSelect} loading={loading} style={{ marginBottom: 16 }}>
            AI 推荐文件
          </Button>

          {aiFiles.length > 0 && (
            <div style={{ border: '1px solid var(--border-default)', borderRadius: 8, padding: 12 }}>
              <div style={{ fontWeight: 500, marginBottom: 8 }}>AI推荐的文件</div>
              {aiFiles.map(f => (
                <div key={f.filename} style={{ display: 'flex', alignItems: 'center', gap: 8, marginBottom: 4 }}>
                  <Checkbox
                    checked={selectedFiles.includes(f.filename)}
                    onChange={e => {
                      if (e.target.checked) {
                        setSelectedFiles([...selectedFiles, f.filename])
                      } else {
                        setSelectedFiles(selectedFiles.filter(n => n !== f.filename))
                      }
                    }}
                  />
                  <span>{f.filename}</span>
                  <span style={{ color: 'var(--text-placeholder)', fontSize: 12 }}>— {f.reason}</span>
                </div>
              ))}
            </div>
          )}
        </>
      )}

      <div style={{ marginTop: 16 }}>
        <div style={{ marginBottom: 8, fontWeight: 500 }}>转交说明（可选）</div>
        <Input.TextArea
          rows={2}
          value={handoverNote}
          onChange={e => setHandoverNote(e.target.value)}
          placeholder="填写转交说明..."
        />
      </div>
    </Modal>
  )
}
```

```typescript
// src/components/Handover/ImportDialog.tsx
import { useState } from 'react'
import { Modal, Input, message, Upload, Spin } from 'antd'
import { InboxOutlined } from '@ant-design/icons'
import { projectService } from '../../services/projectService'

interface ImportDialogProps {
  open: boolean
  onClose: () => void
  onImported: () => void
}

export default function ImportDialog({ open, onClose, onImported }: ImportDialogProps) {
  const [loading, setLoading] = useState(false)
  const [preview, setPreview] = useState<any>(null)
  const [zipPath, setZipPath] = useState<string>('')
  const [projectName, setProjectName] = useState('')

  const handleFileSelect = async (file: File) => {
    // 读取ZIP文件
    setLoading(true)
    try {
      // 通过IPC读取文件路径
      const result = await projectService.previewHandover(zipPath)
      if (result.success && result.data) {
        setPreview(result.data)
        setProjectName(result.data.project.name)
      }
    } catch (err) {
      message.error('无法读取转交包')
    } finally {
      setLoading(false)
    }
  }

  const handleImport = async () => {
    setLoading(true)
    try {
      const result = await projectService.importHandover(zipPath, projectName)
      if (result.success) {
        message.success('导入成功')
        onImported()
        onClose()
      }
    } catch (err) {
      message.error('导入失败')
    } finally {
      setLoading(false)
    }
  }

  return (
    <Modal
      title="导入转交项目"
      open={open}
      onCancel={onClose}
      onOk={handleImport}
      okText="确认导入"
      confirmLoading={loading}
      width={640}
    >
      {!preview ? (
        <Upload.Dragger
          accept=".pmaer.zip"
          showUploadList={false}
          beforeUpload={(file) => {
            setZipPath(file.path || '')
            handleFileSelect(file)
            return false
          }}
        >
          <p className="ant-upload-drag-icon"><InboxOutlined /></p>
          <p className="ant-upload-text">点击或拖拽 .pmaer.zip 文件到此处</p>
        </Upload.Dragger>
      ) : (
        <Spin spinning={loading}>
          <div style={{ marginBottom: 16 }}>
            <div style={{ marginBottom: 4, color: 'var(--text-secondary)' }}>项目名称</div>
            <div style={{ fontWeight: 500 }}>{preview.project.name}</div>
          </div>
          <div style={{ marginBottom: 16 }}>
            <div style={{ marginBottom: 4, color: 'var(--text-secondary)' }}>文件数量</div>
            <div>{preview.files.length} 个文件</div>
          </div>
          <div style={{ marginBottom: 16 }}>
            <div style={{ marginBottom: 4, color: 'var(--text-secondary)' }}>项目阶段</div>
            <div>{preview.project.current_stage}</div>
          </div>
          {preview.handover_note && (
            <div style={{ marginBottom: 16 }}>
              <div style={{ marginBottom: 4, color: 'var(--text-secondary)' }}>转交说明</div>
              <div style={{ fontStyle: 'italic' }}>"{preview.handover_note}"</div>
            </div>
          )}
          <div style={{ marginBottom: 16 }}>
            <div style={{ marginBottom: 4, color: 'var(--text-secondary)' }}>导入后项目名称</div>
            <Input
              value={projectName}
              onChange={e => setProjectName(e.target.value)}
            />
          </div>
        </Spin>
      )}
    </Modal>
  )
}
```
