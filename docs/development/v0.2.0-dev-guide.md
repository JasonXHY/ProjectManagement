```typescript
// 工具函数：获取下一个里程碑
function getNextMilestones(milestones: Milestone[]): Milestone | null {
  if (!milestones || milestones.length === 0) return null;
  const now = new Date();
  const pending = milestones
    .filter(m => new Date(m.date) >= now)
    .sort((a, b) => new Date(a.date).getTime() - new Date(b.date).getTime());
  return pending[0] || null;
}
```

```typescript
// src/components/ProjectHome/MilestoneModal.tsx

import { Modal, Timeline, Tag } from 'antd';

interface MilestoneModalProps {
  open: boolean;
  onClose: () => void;
  milestones: Milestone[];
}

export const MilestoneModal: React.FC<MilestoneModalProps> = ({
  open, onClose, milestones
}) => {
  const now = new Date();

  const getTimelineItems = () => {
    let foundNext = false;
    return milestones.map(m => {
      const mDate = new Date(m.date);
      const isPast = mDate < now;
      const isNext = !foundNext && !isPast;

      if (isNext) foundNext = true;

      return {
        color: isPast ? '#4F46E5' : isNext ? '#F59E0B' : '#E5E7EB',
        dot: m.is_key ? <span style={{ fontSize: 10 }}>★</span> : undefined,
        children: (
          <div style={{ display: 'flex', justifyContent: 'space-between' }}>
            <span style={{
              fontWeight: isNext ? 600 : 400,
              color: isNext ? '#F59E0B' : isPast ? '#6B7280' : '#9CA3AF',
              textDecoration: isPast ? 'line-through' : 'none',
            }}>
              {m.title}
              {isNext && <Tag color="warning" style={{ marginLeft: 8 }}>下一步</Tag>}
            </span>
            <span style={{
              fontSize: 12,
              color: isNext ? '#F59E0B' : '#9CA3AF',
              fontFeatureSettings: '"tnum"',
            }}>
              {m.date}
            </span>
          </div>
        ),
      };
    });
  };

  return (
    <Modal
      open={open}
      title="里程碑时间轴"
      onCancel={onClose}
      footer={null}
      width={640}
    >
      <Timeline items={getTimelineItems()} />
    </Modal>
  );
};
```

```typescript
function getStageGroup(selectedStage: string, projectStage: string): string {
  if (selectedStage === 'all') {
    // 跟随项目当前阶段
    if (projectStage === '售前') return 'presale';
    if (projectStage === '关闭') return 'closed';
    return 'progress';
  }
  if (selectedStage === 'presale') return 'presale';
  if (selectedStage === 'closed') return 'closed';
  return 'progress';  // 启动/需求/方案/构建/测试/上线/验收/转客户成功
}

export const FeatureCards: React.FC<FeatureCardsProps> = ({
  selectedStage, project, files
}) => {
  const group = getStageGroup(selectedStage, project.current_stage);

  const renderCards = () => {
    switch (group) {
      case 'presale':
        return (
          <>
            <ProjectInfoCard project={project} />
            <ContractOverviewCard files={files} />
            <ProjectEvaluationCard project={project} />
          </>
        );
      case 'progress':
        return (
          <>
            <ReqTrackingCard project={project} />
            <KeyIssuesCard project={project} />
            <SignatureTrackingCard files={files} />
          </>
        );
      case 'closed':
        return (
          <>
            <BusinessOppCard project={project} />
            <DeliverablesCard files={files} />
            <ProjectSummaryCard project={project} />
          </>
        );
      default:
        return null;
    }
  };

  return (
    <div className="feature-row">
```

### 4.1 HandoverService 核心逻辑

```typescript
// electron/services/handover-service.ts

import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import JSZip from 'jszip';
import { v4 as uuidv4 } from 'uuid';
import { getProject, getFilesByProject } from '../database/projects';
import { getProjectSummary } from './ai-service';
import { HandoverJSON, HandoverFile } from '../../src/types/handover';

export class HandoverService {

  // 整体转交：打包所有文件
  async exportFull(projectId: string, handoverNote?: string): Promise<string> {
    const project = await getProject(projectId);
    const files = await getFilesByProject(projectId);
    const projectPath = await resolveProjectPath(projectId);

    // 生成 handover.json
    const handoverData = this.buildHandoverJSON(project, files, 'full', handoverNote);

    // 创建临时目录
    const tempDir = path.join(os.tmpdir(), `pmaer-handover-${uuidv4()}`);
    fs.mkdirSync(tempDir, { recursive: true });

    try {
      // 复制文件到临时目录
      for (const file of files) {
        const srcPath = path.join(projectPath, file.category, file.subcategory || '', file.filename);
        const destPath = path.join(tempDir, 'files', file.category, file.subcategory || '', file.filename);

        if (fs.existsSync(srcPath)) {
          fs.mkdirSync(path.dirname(destPath), { recursive: true });
          fs.copyFileSync(srcPath, destPath);
        }
      }

      // 写入 handover.json
      fs.writeFileSync(
        path.join(tempDir, 'handover.json'),
        JSON.stringify(handoverData, null, 2),
        'utf-8'
      );

      // 写入 context.md（如有）
      if (handoverNote) {
        fs.writeFileSync(path.join(tempDir, 'context.md'), handoverNote, 'utf-8');
      }

      // 写入 project-summary.md（如有）
      const summary = await getProjectSummary(projectId);
      if (summary) {
        fs.writeFileSync(path.join(tempDir, 'project-summary.md'), summary, 'utf-8');
      }

      // 打包 ZIP
      const zip = new JSZip();
      this.addDirectoryToZip(zip, tempDir, '');

      const zipBuffer = await zip.generateAsync({ type: 'nodebuffer' });

      // 返回给前端选择保存位置
      return zipBuffer;

    } finally {
      // 清理临时目录
      fs.rmSync(tempDir, { recursive: true, force: true });
    }
  }

  // 协作转交：打包选择的文件
  async exportSelective(
    projectId: string,
    selectedFiles: { filename: string; reason?: string }[],
    handoverNote?: string
  ): Promise<string> {
    // 逻辑与 exportFull 类似，但只包含 selectedFiles 中的文件
    // ...
  }

  // AI推荐文件
  async aiSelectFiles(
    projectId: string,
    description: string
  ): Promise<{ files: { filename: string; reason: string }[]; keyContext: string }> {
    const project = await getProject(projectId);
    const files = await getFilesByProject(projectId);
    const summary = await getProjectSummary(projectId);

    // 构建文件列表（含摘要前100字）
    const fileList = files.map(f => ({
      filename: f.filename,
      category: f.category,
      subcategory: f.subcategory,
      ai_summary: f.ai_summary?.substring(0, 100) || '',
    }));

    // 调用AI
    const prompt = HANDOVER_SELECT_PROMPT
      .replace('{user_description}', description)
      .replace('{file_list_with_categories_and_summaries}', JSON.stringify(fileList, null, 2))
      .replace('{project_summary}', summary || '暂无');

    const response = await chat(projectId, prompt);

    // 解析JSON
    try {
      const result = JSON.parse(response);
      return {
        files: result.selected_files || [],
        keyContext: result.key_context || '',
      };
    } catch {
      return { files: [], keyContext: '' };
    }
  }

  // 导入预览
  async previewImport(zipPath: string): Promise<HandoverJSON> {
    const zip = await JSZip.loadAsync(fs.readFileSync(zipPath));
    const handoverJson = await zip.file('handover.json')?.async('string');

    if (!handoverJson) {
      throw new Error('转交包中缺少 handover.json');
    }

    const data = JSON.parse(handoverJson) as HandoverJSON;

    if (data.format_version !== '1.0') {
      throw new Error(`转交包格式版本 ${data.format_version} 不兼容`);
    }

    return data;
  }

  // 导入项目
  async importProject(zipPath: string, projectName?: string): Promise<string> {
    const data = await this.previewImport(zipPath);
    const zip = await JSZip.loadAsync(fs.readFileSync(zipPath));

    // 创建新项目
    const finalName = projectName || data.project.name;
    const projectId = await createProject({
      name: finalName,
      category_type: data.project.category_type || 'stage',
    });

    const projectPath = await resolveProjectPath(projectId);

    // 解压文件
    for (const [relativePath, file] of Object.entries(zip.files)) {
      if (relativePath.startsWith('files/') && !file.dir) {
        const destPath = path.join(projectPath, relativePath.replace('files/', ''));
        fs.mkdirSync(path.dirname(destPath), { recursive: true });
        const content = await file.async('nodebuffer');
        fs.writeFileSync(destPath, content);
      }
    }

    // 导入元数据
    await updateProject(projectId, {
      current_stage: data.project.current_stage,
      milestones: data.project.milestones,
      metadata: data.project.metadata,
    });

    // 导入文件记录
    for (const file of data.files) {
      await insertFile({
        project_id: projectId,
        filename: file.filename,
        category: file.category,
        subcategory: file.subcategory,
        file_type: file.file_type,
        file_size: file.file_size,
        content_extracted: file.content_extracted,
        signature_status: file.signature_status,
        ai_summary: file.ai_summary,
        ai_key_info: file.ai_key_info,
      });
    }

    return projectId;
  }

  // 辅助方法：构建 handover JSON
  private buildHandoverJSON(project, files, type, note?): HandoverJSON {
    return {
      format_version: '1.0',
      type,
      project: {
        name: project.name,
        category_type: project.category_type,
        current_stage: project.current_stage,
        created_at: project.created_at,
        milestones: project.milestones,
        metadata: project.metadata,
      },
      files: files.map(f => ({
        filename: f.filename,
        relative_path: `files/${f.category}/${f.subcategory || ''}/${f.filename}`,
        category: f.category,
        subcategory: f.subcategory,
        stage: f.stage || project.current_stage,
        file_type: f.file_type,
        file_size: f.file_size,
        content_extracted: f.content_extracted,
        signature_status: f.signature_status,
        ai_summary: f.ai_summary,
        ai_key_info: f.ai_key_info,
      })),
      handover_note: note,
      exported_at: new Date().toISOString(),
      exported_by: 'PMAer v0.2.0',
    };
  }

  // 辅助方法：递归添加目录到ZIP
  private addDirectoryToZip(zip: JSZip, dirPath: string, zipPath: string) {
    const items = fs.readdirSync(dirPath);
    for (const item of items) {
      const fullPath = path.join(dirPath, item);
      const relativePath = zipPath ? `${zipPath}/${item}` : item;

      if (fs.statSync(fullPath).isDirectory()) {
        this.addDirectoryToZip(zip, fullPath, relativePath);
      } else {
        const content = fs.readFileSync(fullPath);
        zip.file(relativePath, content);
      }
    }
  }
}
```

### 4.2 IPC Handler 注册

```typescript
// electron/ipc/handover-handlers.ts

import { ipcMain, dialog, BrowserWindow } from 'electron';
import { HandoverService } from '../services/handover-service';

const handoverService = new HandoverService();

export function registerHandoverHandlers() {
  // 整体/协作转交
  ipcMain.handle('handover:export', async (event, { projectId, mode, selectedFiles, handoverNote }) => {
    try {
      let zipBuffer: Buffer;

      if (mode === 'full') {
        zipBuffer = await handoverService.exportFull(projectId, handoverNote);
      } else {
        zipBuffer = await handoverService.exportSelective(projectId, selectedFiles, handoverNote);
      }

      // 弹出保存对话框
      const win = BrowserWindow.fromWebContents(event.sender);
      const result = await dialog.showSaveDialog(win!, {
        defaultPath: `project-export.pmaer.zip`,
        filters: [{ name: 'PMAer转交包', extensions: ['pmaer.zip'] }],
      });

      if (!result.canceled && result.filePath) {
        fs.writeFileSync(result.filePath, zipBuffer);
        return { success: true, filePath: result.filePath };
      }

      return { success: false };
    } catch (error) {
      return { success: false, error: error.message };
    }
  });

  // AI推荐文件
  ipcMain.handle('handover:ai-select', async (event, { projectId, description }) => {
    try {
      const result = await handoverService.aiSelectFiles(projectId, description);
      return { success: true, ...result };
    } catch (error) {
      return { success: false, files: [], keyContext: '', error: error.message };
    }
  });

  // 预览转交包
  ipcMain.handle('handover:preview', async (event, { zipPath }) => {
    try {
      const data = await handoverService.previewImport(zipPath);
      return { success: true, data };
    } catch (error) {
      return { success: false, error: error.message };
    }
  });

  // 导入项目
  ipcMain.handle('handover:import', async (event, { zipPath, projectName }) => {
    try {
      const projectId = await handoverService.importProject(zipPath, projectName);
      return { success: true, projectId };
    } catch (error) {
      return { success: false, error: error.message };
    }
  });
}
```

```typescript
  const [mode, setMode] = useState<'full' | 'selective'>('full');
  const [description, setDescription] = useState('');
  const [handoverNote, setHandoverNote] = useState('');
  const [aiFiles, setAiFiles] = useState<{ filename: string; reason: string }[]>([]);
  const [selectedFiles, setSelectedFiles] = useState<string[]>([]);
  const [loading, setLoading] = useState(false);

  const handleAiSelect = async () => {
    if (!description.trim()) {
      message.warning('请填写新同事描述');
      return;
    }
    setLoading(true);
    try {
      const result = await window.api.handover.aiSelect(projectId, description);
      if (result.success) {
        setAiFiles(result.files);
        setSelectedFiles(result.files.map(f => f.filename));
      }
    } finally {
      setLoading(false);
    }
  };

  const handleExport = async () => {
    setLoading(true);
    try {
      const result = await window.api.handover.export({
        projectId,
        mode,
        selectedFiles: mode === 'selective' ? selectedFiles : undefined,
        handoverNote,
      });
      if (result.success) {
        message.success('导出成功');
        onClose();
      }
    } finally {
      setLoading(false);
    }
  };

  return (
    <Modal
      open={open}
      title="项目转交"
      onCancel={onClose}
      width={640}
      footer={[
        <Button key="cancel" onClick={onClose}>取消</Button>,
        <Button key="export" type="primary" loading={loading} onClick={handleExport}>
          导出 .pmaer.zip
        </Button>,
      ]}
    >
      <Radio.Group value={mode} onChange={e => setMode(e.target.value)}>
        <Radio value="full">整体转交 — 打包所有项目文件和信息</Radio>
        <Radio value="selective">协作转交 — AI推荐相关文件，可手动调整</Radio>
      </Radio.Group>

      {mode === 'selective' && (
        <>
          <div style={{ marginTop: 16 }}>
            <label>新同事描述 *</label>
            <Input.TextArea
              value={description}
              onChange={e => setDescription(e.target.value)}
              placeholder="例：张工负责前端开发，需要了解UI规范和接口文档"
              rows={3}
            />
          </div>

          <Button
            type="primary"
            icon={<BulbOutlined />}
            onClick={handleAiSelect}
            loading={loading}
            style={{ marginTop: 12 }}
          >
            AI推荐文件
          </Button>

          {aiFiles.length > 0 && (
            <div style={{ marginTop: 16 }}>
              <div style={{ fontWeight: 600, marginBottom: 8 }}>AI推荐的文件：</div>
              {aiFiles.map(f => (
                <div key={f.filename} style={{ display: 'flex', gap: 8, marginBottom: 4 }}>
                  <Checkbox
                    checked={selectedFiles.includes(f.filename)}
                    onChange={e => {
                      if (e.target.checked) {
                        setSelectedFiles([...selectedFiles, f.filename]);
                      } else {
                        setSelectedFiles(selectedFiles.filter(n => n !== f.filename));
                      }
                    }}
                  >
                    {f.filename}
                  </Checkbox>
                  <span style={{ color: '#6B7280', fontSize: 12 }}>{f.reason}</span>
                </div>
              ))}
            </div>
          )}
        </>
      )}

      <div style={{ marginTop: 16 }}>
        <label>转交说明（可选）</label>
        <Input.TextArea
          value={handoverNote}
          onChange={e => setHandoverNote(e.target.value)}
          placeholder="填写转交说明..."
          rows={3}
        />
      </div>
    </Modal>
  );
};
```

## 七、风险与应对

| 风险 | 影响 | 应对 |
|------|------|------|
| AI文件推荐不准确 | 协作转交体验差 | 提供手动调整入口，AI只辅助不决定 |
| 转交包体积过大 | 传输困难 | 协作转交本身精简；整体转交显示预估大小 |
| Markdown渲染XSS | 安全风险 | rehype-sanitize过滤危险标签 |
| 跨版本兼容 | 旧版无法导入新版 | format_version字段校验+向后兼容 |
| 同名项目冲突 | 导入时覆盖 | 自动重命名+用户确认 |
